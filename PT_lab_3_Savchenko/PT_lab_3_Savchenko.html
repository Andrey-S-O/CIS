def kotelnikov_series(y_n, T_a, t):
    restored_signal = np.zeros_like(t)
    for n in range(len(y_n)):
        restored_signal += y_n[n] * np.sinc((t - n * T_a) / T_a)
    return restored_signal

#       y[n]
y_t_restored = kotelnikov_series(y_n, T_a, t)
